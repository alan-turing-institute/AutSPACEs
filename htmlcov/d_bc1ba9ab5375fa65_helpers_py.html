10def user_profile_exists(user): 

11 """ 

12 Determines whether the user profile for the user exists. If it doesn't 

13 exist, it means that this is a first-time login 

14 

15 A default profile is created immediately when entering the user profile 

16 page, so this will return True until that point and False afterwards. 

17 

18 Args: 

19 user: request.user 

20 

21 Returns: 

22 True if this is the first time the user has logged in, False o/w. 

23 """ 

24 uo = UserProfile.objects.filter(user=user) 

25 return (uo.count() > 0)

27def user_submitted_profile(user): 

28 """ 

29 Determines whether the user has ever hit the Submit button on the profile page. 

30 

31 This is the best indication we have of whether the user has ever filled out 

32 the profile page (because they might legitimately want the profile to be 

33 emtpy). 

34 

35 Args: 

36 user: request.user 

37 

38 Returns: 

39 True if the user has ever submitted a profile update, False o/w. 

40 """ 

41 try: 

42 submitted = UserProfile.objects.get(user=user).profile_submitted 

43 except UserProfile.DoesNotExist: 

44 submitted = False 

45 return submitted

63def delete_user(user, delete_oh_data): 

64 """ 

65 Deletes the user and all data associated with it. 

66 

67 Args: 

68 delete_oh_data: True if stories on OpenHumans should also be deleted 

69 

70 Returns: 

71 None 

72 """ 

73 ohmember = user.openhumansmember 

74 

75 # Delete the stories from the OpenHumans database 

76 if delete_oh_data: 

77 ohmember.delete_all_files() 

78 

79 # de-auth OH member if not doing so by default 

80 deauth_on_delete = getattr(settings, 'OPENHUMANS_DEAUTH_ON_DELETE', True) 

81 if deauth_on_delete == False: 

82 ohmember.deauth() 

83 

84 # Delete the actual user 

85 user.delete()
