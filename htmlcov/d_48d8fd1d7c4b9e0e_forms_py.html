206 def clean(self): 

207 # Raise an error if status is "not reviewed" and moderation_comments isn't empty 

208 cleaned_data = super().clean() 

209 moderation_status = cleaned_data.get("moderation_status") 

210 moderation_reply = cleaned_data.get("moderation_reply") 

211 if moderation_status == "not reviewed" and moderation_reply: 

212 raise ValidationError( 

213 "Stories that are not being reviewed cannot have moderation replies added to them." 

214 ) 

215 # Raise an error if status is "approved" and there are "Red" severity moderation comments 

216 if moderation_status == "approved": 

217 try: 

218 reasons = json.loads(moderation_reply) 

219 red_reasons = sum(map(lambda x : 1 if x.get('severity', None) == 'red' else 0, reasons)) 

220 except: 

221 red_reasons = 0 

222 if red_reasons > 0: 

223 raise ValidationError( 

224 "Stories with Red moderation reasons can't be given approved status."
